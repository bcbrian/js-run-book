## How to Keep it DRY(Dont Repeat Yourself)

### Functions are a way

```js
// basic
function mySpecialFunction() {
  // do something cool!
}
// returns will stop execution
function returns(){
  // some code here
  alert("You will see me");
  return;
  alert("You wont see me");
  // and code here won't run
}

// returns will send info back if provided
function sum(oneParam, twoParam){
  return oneParam + twoParam
}
// this way 
const six = sum(3, 3)
alert(six); // 6
// this way ... samesies
let oneParam = 3;
let twoParam = 3;
const answer = sum(oneParam, twoParam)
alert(answer); // 6

let oneParam = 21;
let twoParam = 21;
const answerTwo = sum(oneParam, twoParam)
alert(answer); // 42 it is always the right answer...

```
